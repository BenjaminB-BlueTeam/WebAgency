# Pappers Enrichment Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enrich prospect data with legal/financial info (SIRET, date de création, CA, effectifs) from the Pappers API during scoring, and display it in the prospect info tab.

**Architecture:** A pure `enrichWithPappers(nom, ville)` function fetches Pappers data silently (returns `null` on any error). The score endpoint calls it before `scoreProspect`, passes results to `scoreFinancier` for an enriched Claude prompt, and persists the 4 Pappers fields if enrichment succeeded. The UI shows a conditional "Données légales" section in the prospect info tab.

**Tech Stack:** TypeScript, Prisma (SQLite/Turso), Next.js App Router, Vitest

```ts
export type PappersData = {
  siret: string
  dateCreation: Date
  chiffreAffaires: number | null
  effectifs: string | null
}

export async function enrichWithPappers(
  nom: string,
  ville: string
): Promise<PappersData | null> {
  const apiKey = process.env.PAPPERS_API_KEY
  if (!apiKey) return null

  try {
    const url = new URL("https://api.pappers.fr/v2/recherche")
    url.searchParams.set("q", nom)
    url.searchParams.set("ville", ville)
    url.searchParams.set("api_token", apiKey)

    const res = await fetch(url.toString())
    if (!res.ok) return null

    const data = await res.json()
    const resultats = data.resultats as Array<{
      siret?: string
      date_creation?: string
      chiffre_affaires?: number
      tranche_effectif?: string
    }>

    if (!resultats || resultats.length === 0) return null

    const r = resultats[0]
    if (!r.siret || !r.date_creation) return null

    return {
      siret: r.siret,
      dateCreation: new Date(r.date_creation),
      chiffreAffaires: r.chiffre_affaires ?? null,
      effectifs: r.tranche_effectif ?? null,
    }
  } catch {
    console.error("[pappers] enrichWithPappers silenced error:", nom, ville)
    return null
  }
}
```

export async function scoreFinancier(
  activite: string,
  ville: string,
  noteGoogle: number | null,
  nbAvisGoogle: number | null,
  pappersData?: PappersData | null
): Promise<number | null> {
  try {
    const pappersContext = pappersData
      ? `\n- Ancienneté : ${new Date().getFullYear() - pappersData.dateCreation.getFullYear()} ans\n- CA : ${pappersData.chiffreAffaires !== null ? pappersData.chiffreAffaires + " €" : "inconnu"}\n- Effectifs : ${pappersData.effectifs ?? "inconnus"}`
      : ""
    const response = await analyzeWithClaude(
      "Tu es un expert en prospection commerciale pour les agences web. Réponds uniquement en JSON valide.",
      `Basé sur ces données : activité = ${activite}, ville = ${ville}, note Google = ${noteGoogle ?? "inconnue"}/5, nombre d'avis = ${nbAvisGoogle ?? "inconnu"}.${pappersContext} Estime de 0 à 10 si cette entreprise a la capacité financière d'investir ~2000€ dans un site web. Considère le secteur d'activité, le volume d'activité estimé via les avis, et la zone géographique. Réponds uniquement en JSON : {"score": number, "justification": string}`
    )
    const parsed = parseClaudeJSON<{ score: number }>(response)
    return Math.max(0, Math.min(10, parsed.score))
  } catch {
    return null
  }

```ts
import { NextRequest, NextResponse } from "next/server"
import { requireAuth } from "@/lib/auth"
import { prisma } from "@/lib/db"
import { scoreProspect } from "@/lib/scoring"
import { enrichWithPappers } from "@/lib/pappers"

type RouteParams = { params: Promise<{ id: string }> }

export async function POST(request: NextRequest, { params }: RouteParams) {
  try {
    await requireAuth()

    const { id } = await params

    const prospect = await prisma.prospect.findUnique({
      where: { id },
      select: {
        id: true,
        nom: true,
        siteUrl: true,
        activite: true,
        ville: true,
        noteGoogle: true,
        nbAvisGoogle: true,
      },
    })

    if (!prospect) {
      return NextResponse.json(
        { error: "Prospect non trouvé" },
        { status: 404 }
      )
    }

    const pappersData = await enrichWithPappers(prospect.nom, prospect.ville).catch(() => null)

    const scores = await scoreProspect(
      {
        siteUrl: prospect.siteUrl,
        activite: prospect.activite,
        ville: prospect.ville,
        noteGoogle: prospect.noteGoogle,
        nbAvisGoogle: prospect.nbAvisGoogle,
      },
      pappersData
    )

    await prisma.prospect.update({
      where: { id },
      data: {
        scorePresenceWeb: scores.scorePresenceWeb,
        scoreSEO: scores.scoreSEO,
        scoreDesign: scores.scoreDesign,
        scoreFinancier: scores.scoreFinancier,
        scorePotentiel: scores.scorePotentiel,
        scoreGlobal: scores.scoreGlobal,
        ...(pappersData
          ? {
              siret: pappersData.siret,
              dateCreation: pappersData.dateCreation,
              chiffreAffaires: pappersData.chiffreAffaires,
              effectifs: pappersData.effectifs,
            }
          : {}),
      },
    })

    await prisma.activite.create({
      data: {
        prospectId: id,
        type: "SCORING",
        description: `Scoring effectué — Score global : ${scores.scoreGlobal ?? "N/A"}/10`,
      },
    })

    return NextResponse.json({ data: scores })
  } catch (error) {
    if (error instanceof Error && error.message === "Unauthorized") {
      return NextResponse.json({ error: "Non autorisé" }, { status: 401 })
    }
    return NextResponse.json({ error: "Erreur serveur" }, { status: 500 })
  }
}
```

              <div className="flex items-start gap-2">
                <Building2 size={14} className="mt-0.5 shrink-0 text-[#555555]" />
                <div className="min-w-0">
                  <p className="text-xs text-[#555555]">Création</p>
                  <p className="text-sm text-[#fafafa]">
                    {new Date(prospect.dateCreation).toLocaleDateString("fr-FR")}
                    <span className="ml-1 text-[#737373]">
                      ({new Date().getFullYear() - new Date(prospect.dateCreation).getFullYear()} ans)
                    </span>
                  </p>
                </div>
              </div>
            )}
            {prospect.chiffreAffaires !== null && (
              <div className="flex items-start gap-2">
                <Building2 size={14} className="mt-0.5 shrink-0 text-[#555555]" />
                <div className="min-w-0">
                  <p className="text-xs text-[#555555]">Chiffre d&apos;affaires</p>
                  <p className="text-sm text-[#fafafa]">
                    {new Intl.NumberFormat("fr-FR").format(prospect.chiffreAffaires)} €
                  </p>
                </div>
              </div>
            )}
            {prospect.effectifs && (
              <div className="flex items-start gap-2">
                <Building2 size={14} className="mt-0.5 shrink-0 text-[#555555]" />
                <div className="min-w-0">
                  <p className="text-xs text-[#555555]">Effectifs</p>
                  <p className="text-sm text-[#fafafa]">{prospect.effectifs} salariés</p>
                </div>
              </div>
            )}
          </div>
        </div>
      )}
